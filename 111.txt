ОСНОВНЫЕ ПРИНЦИПЫ ПРЕДОТВРАЩЕНИЯ
Принцип 1: Всегда используйте using для ресурсов
Любой объект, реализующий IDisposable, должен быть обернут в using.
// Вместо этого:
FileStream stream = File.OpenRead("file.txt");
// работа с stream
// stream не закрыт

// Делайте так:
using (FileStream stream = File.OpenRead("file.txt"))
{
    // работа с stream
} // stream автоматически закроется

// Или так (короткая форма):
using FileStream stream = File.OpenRead("file.txt");
// работа с stream
// stream закроется здесь
Какие ресурсы требуют using:
●	FileStream, StreamReader, StreamWriter
●	SqlConnection, SqlCommand
●	HttpClient
●	Bitmap, Image
●	Timer
●	Любой класс с IDisposable
Принцип 2: Отписывайтесь от событий
Любая подписка на событие должна иметь соответствующую отписку.
// Вместо этого:
public class Window
{
    public Window(Button button)
    {
        button.Click += OnButtonClick; // подписались
    }
    
    void OnButtonClick(object sender, EventArgs e)
    {
        // обработка
    }
}

// Делайте так:
public class Window : IDisposable
{
    private Button _button;
    
    public Window(Button button)
    {
        _button = button;
        _button.Click += OnButtonClick; // подписались
    }
    
    void OnButtonClick(object sender, EventArgs e)
    {
        // обработка
    }
    
    public void Dispose()
    {
        _button.Click -= OnButtonClick; // отписались
    }
}
Правила отписки:
●	Отписывайтесь в том же классе, где подписались
●	Если класс может быть уничтожен - реализуйте IDisposable
●	В формах отписывайтесь в FormClosing или Dispose
Принцип 3: Контролируйте статические коллекции
Статические коллекции живут вечно. Они должны иметь механизмы очистки.
// Вместо этого:
public static class Cache
{
    public static List<Data> Items = new List<Data>(); // будет расти вечно
}

// Делайте так:
public static class Cache
{
    private static List<Data> _items = new List<Data>();
    private static readonly int _maxSize = 100;
    
    public static void Add(Data item)
    {
        if (_items.Count >= _maxSize)
        {
            _items.RemoveAt(0); // удаляем старые
        }
        _items.Add(item);
    }
    
    public static void Clear()
    {
        _items.Clear(); // возможность ручной очистки
    }}
Альтернативы статическим коллекциям:
●	Используйте WeakReference для некритичных данных
●	Добавляйте время жизни (TTL) для элементов кеша
●	Рассмотрите использование MemoryCache из .NET
Принцип 4: Не храните большие объекты дольше нужного
Если большой объект не нужен - обнулите ссылку на него.
// Вместо этого:
byte[] bigData = LoadHugeFile();
Process(bigData);
// bigData все еще висит в памяти

// Делайте так:
byte[] bigData = LoadHugeFile();
Process(bigData);
bigData = null; // разрешаем сборщику мусора забрать данные

// Или лучше - ограничьте область видимости:
void ProcessFile()
{
    byte[] bigData = LoadHugeFile();
    Process(bigData);
} // bigData выйдет из области видимости
Принцип 5: Будьте осторожны с кэшированием
Кэш должен иметь политику вытеснения.
// Вместо этого:
public class ImageCache
{
    private Dictionary<string, Bitmap> _cache = new(); // будет расти вечно
}

// Делайте так:
public class ImageCache
{
    private Dictionary<string, CacheItem> _cache = new();
    private readonly int _maxItems = 50;
    private readonly TimeSpan _ttl = TimeSpan.FromMinutes(10);
    
    public void Add(string key, Bitmap image)
    {
        if (_cache.Count >= _maxItems)
        {
            RemoveOldest();
        }
        
        _cache[key] = new CacheItem 
        { 
            Image = image, 
            CreatedAt = DateTime.Now 
        };
    }
    
    private void RemoveOldest()
    {
        var oldest = _cache.OrderBy(x => x.Value.CreatedAt).First();
        _cache.Remove(oldest.Key);
    }
    
    private void RemoveExpired()
    {
        var expired = _cache
            .Where(x => DateTime.Now - x.Value.CreatedAt > _ttl)
            .Select(x => x.Key)
            .ToList();
            
        foreach (var key in expired)
        {
            _cache.Remove(key);
        }
    }
    
    private class CacheItem
    {
        public Bitmap Image { get; set; }
        public DateTime CreatedAt { get; set; }
    }
}
Принцип 6: Освобождайте ресурсы в обратном порядке
При использовании нескольких вложенных ресурсов освобождайте их в порядке, обратном созданию.
// Правильно:
using (var connection = new SqlConnection(connectionString))
{
    connection.Open();
    
    using (var command = new SqlCommand(query, connection))
    {
        using (var reader = command.ExecuteReader())
        {
            // работаем с reader
        } // 3. закрывается reader
    } // 2. закрывается command
} // 1. закрывается connection

// Или одной строкой:
using var connection = new SqlConnection(connectionString);
using var command = new SqlCommand(query, connection);
connection.Open();
using var reader = command.ExecuteReader();
// работаем - всё освободится при выходе из метода
Принцип 7: Проверяйте утечки до того, как они произойдут
Добавьте мониторинг памяти в критических местах.
public static class MemoryGuard
{
    [Conditional("DEBUG")]
    public static void Check(string location)
    {
        GC.Collect();
        GC.WaitForPendingFinalizers();
        
        long memory = GC.GetTotalMemory(false);
        Console.WriteLine($"[{location}] Память: {memory / 1024 / 1024} MB");
        
        _lastCheck[location] = memory;
    }
    
    private static Dictionary<string, long> _lastCheck = new();
    
    [Conditional("DEBUG")]
    public static void Compare(string location)
    {
        if (_lastCheck.TryGetValue(location, out var previous))
        {
            long current = GC.GetTotalMemory(false);
            long diff = current - previous;
            
            if (diff > 10 * 1024 * 1024) // больше 10 MB
            {
                Console.WriteLine($"Потенциальная утечка в {location}: +{diff / 1024 / 1024} MB");
            }
        }
    }
}

// Использование:
void ProcessData()
{
    MemoryGuard.Check("Начало обработки");
    
    // ... код ...
    
    MemoryGuard.Compare("Начало обработки");
}
Принцип 8: Избегайте финализаторов, если не уверены
Финализаторы замедляют сборку мусора и могут создавать проблемы.
// Плохо: финализатор без необходимости
public class MyClass
{
    ~MyClass()
    {
        // ничего важного не делаем
    }
}

// Хорошо: финализатор только когда действительно нужно
public class SafeHandleClass : IDisposable
{
    private IntPtr _handle;
    
    public void Dispose()
    {
        ReleaseHandle();
        GC.SuppressFinalize(this);
    }
    
    ~SafeHandleClass()
    {
        ReleaseHandle();
    }
    
    private void ReleaseHandle()
    {
        if (_handle != IntPtr.Zero)
        {
            // освобождение неуправляемого ресурса
            _handle = IntPtr.Zero;
        }
    }
}
Принцип 9: В формах и окнах отвязывайте данные
При закрытии формы отвязывайте источники данных.
public partial class MainForm : Form
{
    private List<Customer> _customers;
    private Timer _refreshTimer;
    
    protected override void OnFormClosing(FormClosingEventArgs e)
    {
        // Отвязываем данные
        dataGridView1.DataSource = null;
        
        // Останавливаем таймеры
        _refreshTimer?.Stop();
        _refreshTimer?.Dispose();
        
        // Отписываемся от событий
        button1.Click -= Button1_Click;
        
        // Очищаем коллекции
        _customers?.Clear();
        _customers = null;
        
        base.OnFormClosing(e);
    }
}


Перед тем как залить код, проверьте:
●	Все ли открытые файлы обернуты в using?
●	Все ли подписки на события имеют отписки?
●	Есть ли статические коллекции без ограничений?
●	Не хранятся ли большие объекты дольше нужного?
●	Реализован ли IDisposable правильно?
●	Есть ли финализаторы без необходимости?
●	В формах - отвязаны ли данные при закрытии?
●	Таймеры - остановлены и уничтожены?
Итог
Лучшая утечка - та, которой не случилось.
Профилактика занимает 5 минут. Поиск и исправление утечки - несколько часов.
